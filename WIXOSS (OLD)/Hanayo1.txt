#TouhouDanmakufu[Single]
#ScriptVersion[3]
#Title["Hanayo Non-Spell 1"]
#Text["Hanayo Non-Spell 1"]
#Background["script/default_system/Default_Background_IceMountain.txt"]

#include"script/default_system/Default_ShotConst.txt"

let objBoss;
let objShot;
let shotSE = "SFX/SFX/ATTACK4.wav";
let explodeSE = "SFX/SFX/ATTACK3.wav";

@Event{
		alternative(GetEventType())
	case(EV_REQUEST_LIFE){
		SetScriptResult(5000);
        }
	case(EV_REQUEST_TIMER){
		SetScriptResult(45);
	}
	case(EV_REQUEST_SPELL_SCORE){
		SetScriptResult(600000);
	}
}

@Initialize{
	objBoss = ObjEnemy_Create(OBJ_ENEMY_BOSS);
	ObjEnemy_Regist(objBoss);
	SetCommonData("bossActive", 1);
	
	TDrawLoop;
	
	ObjMove_SetDestAtFrame(objBoss, GetCenterX, 120, 60);
	
	LoadSound(shotSE);
	LoadSound(explodeSE);
	
	TFinalize;
	MainTask;
}

task TFinalize {
	while(ObjEnemy_GetInfo(objBoss,INFO_LIFE)>0){yield;}
	Obj_Delete(objBoss);
	DeleteShotAll(TYPE_ALL,TYPE_IMMEDIATE);
	SetAutoDeleteObject(true);
	CloseScript(GetOwnScriptID());
	return;
}

@MainLoop{
	ObjEnemy_SetIntersectionCircleToShot(objBoss,ObjMove_GetX(objBoss),ObjMove_GetY(objBoss),40);
	ObjEnemy_SetIntersectionCircleToPlayer(objBoss,ObjMove_GetX(objBoss),ObjMove_GetY(objBoss),40);
	
	yield;
}

function GetCenterX{
     return GetStgFrameWidth/2;
 }
function GetCenterY{
     return GetStgFrameHeight/2;
 }
function wait(n){loop(n){yield;}}
 
 task TDrawLoop {
	let imgExRumia = GetCurrentScriptDirectory ~ "Hanayo.png";
	ObjPrim_SetTexture(objBoss, imgExRumia);
	ObjSprite2D_SetSourceRect(objBoss, 1, 1, 70, 118);
	ObjSprite2D_SetDestCenter(objBoss);
 }
 
 task MainTask {
	 
	wait(180);
	
	while(ObjEnemy_GetInfo(objBoss,INFO_LIFE)>0){
		movement;
		let angleR = rand(45,135);
		ShotA(angleR);			
		wait(180);
	}
 }
 
 task ShotA(angle) {
	 
	 let obj = CreateShotA1(ObjMove_GetX(objBoss),ObjMove_GetY(objBoss),4,angle,165,15);
	 PlaySE(shotSE);
	 
	 let objTimer = 0;
	 
	while (!Obj_IsDeleted(obj)) {
		let angleR = rand(0,360);
		CreateShotA1(ObjMove_GetX(obj),ObjMove_GetY(obj),rand(1,2),angleR,10,5);
		angleR = rand(0,360);
		CreateShotA1(ObjMove_GetX(obj),ObjMove_GetY(obj),rand(0.5,1),angleR,3,0);
		wait(3);
		objTimer++;	
		if (objTimer == 30) {
			let angleC = rand(0,360);
			PlaySE(explodeSE);
			loop(28) {
				CreateShotA1(ObjMove_GetX(obj),ObjMove_GetY(obj),3,angleC,65,30);
				CreateShotA1(ObjMove_GetX(obj),ObjMove_GetY(obj),2,angleC,66,20);
				CreateShotA1(ObjMove_GetX(obj),ObjMove_GetY(obj),1,angleC,67,10);
				angleC+=360/28;
			}
			Obj_Delete(obj);
		}
	}
	
 }
 
task movement {
    wait(90);
	ObjMove_SetDestAtFrame(objBoss, rand(GetCenterX+100,GetCenterX-100), rand(GetCenterY-50,GetCenterY-160), 45);
}
